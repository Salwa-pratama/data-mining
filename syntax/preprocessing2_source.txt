# CELL 40
def decimal_scaling(df, cols):
    df_scaled = df.copy()

    for col in cols:
        max_val = df_scaled[col].abs().max()
        j = len(str(int(max_val)))
        df_scaled[col] = df_scaled[col] / (10 ** j)

    return df_scaled
